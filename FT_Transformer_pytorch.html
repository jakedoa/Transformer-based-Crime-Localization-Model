def plot_confusion_matrix(cm, names, title='Confusion matrix', cmap=plt.cm.Blues):
    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(names))
    plt.xticks(tick_marks, names, rotation=45)
    plt.yticks(tick_marks, names)
    plt.tight_layout()
    plt.ylabel('True label')
    plt.xlabel('Predicted label')

# Plot ROC curves for multiclass classification
def plot_multiclass_roc(y_true, y_prob, class_names, title='Multiclass ROC Curve'):
    '''y_true: True labels (1D array of integers)
       y_prob: Predicted probabilities (2D array: [n_samples, n_classes])
       class_names: List of class names'''
    
    n_classes = len(class_names)
    
    # Binarize the labels for ROC computation
    y_true_bin = label_binarize(y_true, classes=list(range(n_classes)))
    
    plt.figure(figsize=(14, 10))
    # Use colormap for multiple classes
    colors = plt.cm.tab20(np.linspace(0, 1, n_classes))
    
    # Store all AUC values for micro/macro average
    fpr_dict = {}
    tpr_dict = {}
    roc_auc_dict = {}

    # Compute ROC curve and AUC for each class
    for i in range(n_classes):
        fpr_dict[i], tpr_dict[i], _ = roc_curve(y_true_bin[:, i], y_prob[:, i])
        roc_auc_dict[i] = auc(fpr_dict[i], tpr_dict[i])
        
        # Plot with some transparency for overlapping curves
        plt.plot(fpr_dict[i], tpr_dict[i], color=colors[i], lw=2,
                 label=f'{class_names[i]} (AUC = {roc_auc_dict[i]:.2f})')
    
    plt.plot([0, 1], [0, 1], 'k--')
    # Formatting
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title(title)
    plt.legend(loc="lower right")
    plt.show()

def apply_model(batch: Dict[str, Tensor]) -> Tensor:
    # Pass None for x_cont since we only have categorical features
    return model(None, batch["x_cat"])

#Loss function of categorical crossentropy
loss_fn = F.cross_entropy

@torch.no_grad()
def evaluate(part: str) -> float:
    model.eval()
    eval_batch_size = 8096

    # Get logits
    y_pred_logits = torch.cat([
        apply_model(batch)
        for batch in delu.iter_batches(data[part], eval_batch_size)
    ])
    
    # Get predictions
    pred = y_pred_logits.argmax(dim=1).cpu().numpy()
    
    # Get probabilities (apply softmax)
    y_pred_probs = F.softmax(y_pred_logits, dim=1).cpu().numpy()
    
    y_true = data[part]["y"].cpu().numpy()
    score = accuracy_score(y_true, pred)
    
    return score, pred, y_true, y_pred_probs
